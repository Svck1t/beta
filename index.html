Configurar sincronización

Conecta esta app a una hoja de Google para que tus datos se sincronicen entre dispositivos. Es un proceso de una sola vez por dispositivo.

1Crea una hoja de cálculo

Ve a sheets.google.com y crea una hoja nueva. Ponle el nombre que quieras, por ejemplo Control Compras.

2Abre Apps Script

Dentro de la hoja, ve al menú Extensiones → Apps Script. Se abre un editor de código en una nueva pestaña.

3Pega el código del servidor

Borra todo lo que aparece por defecto en el editor y pega este código completo:

const SHEET_NAME = 'Compras';
const HEADERS = ['id','fecha','proveedor','producto','cantidad','unidad','costoUnitario','estado','referencia','notas','creadoEn','actualizadoEn'];

function getSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
    sheet.appendRow(HEADERS);
    sheet.getRange('A:B').setNumberFormat('@');
    sheet.getRange('H:J').setNumberFormat('@');
    sheet.setFrozenRows(1);
    sheet.getRange(1, 1, 1, HEADERS.length).setFontWeight('bold');
  }
  return sheet;
}

function doGet(e) {
  try {
    const sheet = getSheet();
    const data = sheet.getDataRange().getValues();
    if (data.length <= 1) return jsonOut({ ok: true, entries: [] });
    const headers = data[0];
    const entries = data.slice(1).map(row => {
      const obj = {};
      headers.forEach((h, i) => {
        let val = row[i];
        if (val instanceof Date) val = Utilities.formatDate(val, Session.getScriptTimeZone(), 'yyyy-MM-dd');
        obj[h] = val;
      });
      return obj;
    });
    return jsonOut({ ok: true, entries: entries });
  } catch (err) { return jsonOut({ ok: false, error: err.toString() }); }
}

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    const action = body.action;
    const sheet = getSheet();
    if (action === 'add') {
      const row = HEADERS.map(h => body.entry[h] !== undefined ? body.entry[h] : '');
      sheet.appendRow(row);
      return jsonOut({ ok: true, entry: body.entry });
    }
    if (action === 'update') {
      const rowIdx = findRowById(sheet, body.entry.id);
      if (rowIdx === -1) return jsonOut({ ok: false, error: 'No encontrado' });
      const row = HEADERS.map(h => body.entry[h] !== undefined ? body.entry[h] : '');
      sheet.getRange(rowIdx, 1, 1, HEADERS.length).setValues([row]);
      return jsonOut({ ok: true, entry: body.entry });
    }
    if (action === 'delete') {
      const rowIdx = findRowById(sheet, body.id);
      if (rowIdx === -1) return jsonOut({ ok: false, error: 'No encontrado' });
      sheet.deleteRow(rowIdx);
      return jsonOut({ ok: true });
    }
    if (action === 'bulk_add') {
      const rows = body.entries.map(entry => HEADERS.map(h => entry[h] !== undefined ? entry[h] : ''));
      if (rows.length > 0) sheet.getRange(sheet.getLastRow() + 1, 1, rows.length, HEADERS.length).setValues(rows);
      return jsonOut({ ok: true, count: rows.length });
    }
    return jsonOut({ ok: false, error: 'Acción desconocida' });
  } catch (err) { return jsonOut({ ok: false, error: err.toString() }); }
}

function findRowById(sheet, id) {
  const data = sheet.getRange(2, 1, Math.max(sheet.getLastRow() - 1, 1), 1).getValues();
  for (let i = 0; i < data.length; i++) if (String(data[i][0]) === String(id)) return i + 2;
  return -1;
}

function jsonOut(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}

Guarda con Ctrl+S y ponle un nombre al proyecto (lo que quieras).

4Despliega como aplicación web

Arriba a la derecha, click en Implementar → Nueva implementación. En la ventana que aparece:

  • Click en el ícono de engranaje y selecciona Aplicación web
  • "Ejecutar como": Yo (tu correo)
  • "Quién tiene acceso": Cualquier usuario
  • Click Implementar

La primera vez te pedirá autorizar. Acepta los permisos. Si ves "Google no ha verificado esta app", click en Configuración avanzada → Ir a [nombre] (no seguro). Es tu propio script, está bien.

Al final aparece una URL larga que empieza con https://script.google.com/macros/s/... — cópiala.

5Pega la URL aquí
Seguridad: cualquiera con esa URL puede ver y modificar tu hoja. Guárdala como una contraseña. Si necesitas revocarla, crea una nueva implementación en Apps Script.
